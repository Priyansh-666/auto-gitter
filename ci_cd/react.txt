#triggers

on:
  push:
    branches:[main, master]
  pull_request:
    branches:[main, master]

#functions

jobs:

  # build job

    build:

      # Name

      name: Build 

      # The OS to run the job on

      runs-on: ubuntu-latest

      # Steps
      steps:
        - name: Checkout  code 

          uses: actions/checkout@v2

        - name: Install node 16

          uses: actions/setup-node@v1

          with:
            node-version: 16.x

        - name: Install NPM Dependencies
          
          run: npm i 

        - name: Build Project
          run: npm run build
